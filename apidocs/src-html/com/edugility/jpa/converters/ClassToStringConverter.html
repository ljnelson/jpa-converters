028package com.edugility.jpa.converters;
029
030import javax.persistence.AttributeConverter;
031import javax.persistence.Converter;
032
033/**
034 * An {@link AttributeConverter} that converts a {@link Class} to and
035 * from its {@linkplain Class#getName() <code>String</code>
036 * representation}.
037 *
038 * @author <a href="http://about.me/lairdnelson"
039 * target="_parent">Laird Nelson</a>
040 *
041 * @see AttributeConverter
042 *
043 * @see Class#toString()
044 */
045@Converter
046public class ClassToStringConverter implements AttributeConverter<Class<?>, String> {
047
048  /**
049   * Creates a new {@link ClassToStringConverter}.
050   */
051  public ClassToStringConverter() {
052    super();
053  }
054
055  /**
056   * Calls the {@link Class#getName()} method on the supplied {@link
057   * Class} and returns its result.
058   *
059   * <p>This method may return {@code null}.</p>
060   *
061   * @param cls the {@link Class} whose {@link String} representation
062   * is desired; may be {@code null} in which case {@code null} will
063   * be returned
064   *
065   * @return the result of calling the {@link Class#getName()} method
066   * on the supplied {@link Class}, or {@code null}
067   */
068  @Override
069  public String convertToDatabaseColumn(final Class<?> cls) {
070    if (cls == null) {
071      return null;
072    }
073    return cls.getName();
074  }
075
076  /**
077   * Calls the {@link #loadClass(String)} method passing it the
078   * supplied {@code className} and returns its result.
079   *
080   * @param className the {@linkplain Class#getName() name} of the
081   * {@link Class} to load; may be {@code null}
082   *
083   * @return a {@link Class} corresponding to the supplied {@code
084   * className}, or {@code null}
085   *
086   * @see #loadClass(String)
087   */
088  @Override
089  public Class<?> convertToEntityAttribute(final String className) {
090    try {
091      return this.loadClass(className);
092    } catch (final ClassNotFoundException cnfe) {
093      throw new IllegalArgumentException(className, cnfe);
094    }
095  }
096
097  /**
098   * Loads a {@link Class} {@linkplain Class#getName() named} by the
099   * supplied {@code name} and returns it.
100   *
101   * <p>This method may return {@code null}.</p>
102   *
103   * <p>This implementation first checks to see if the supplied {@code
104   * name} is {@code null}.  If it is, then {@code null} is returned.
105   * Otherwise, the {@link Class#forName(String, boolean,
106   * ClassLoader)} method is called with the supplied {@code name},
107   * {@code true} and the result of calling the {@link
108   * Thread#getContextClassLoader()} method on the {@linkplain
109   * Thread#currentThread() current <code>Thread</code>} as arguments,
110   * and its result is returned.</p>
111   *
112   * @param name the {@linkplain Class#getName() name} of the {@link
113   * Class} to load; may be {@code null} in which case {@code null}
114   * will be returned
115   *
116   * @return a {@link Class} corresponding to the supplied {@code
117   * name}, or {@code null}
118   *
119   * @exception ClassNotFoundException if the {@link
120   * Class#forName(String, boolean, ClassLoader)} method threw a
121   * {@link ClassNotFoundException}
122   *
123   * @see Class#forName(String, boolean, ClassLoader)
124   *
125   * @see Thread#getContextClassLoader()
126   */
127  protected Class<?> loadClass(final String name) throws ClassNotFoundException {
128    final Class<?> returnValue;
129    if (name == null) {
130      returnValue = null;
131    } else {
132      returnValue = Class.forName(name, true, Thread.currentThread().getContextClassLoader());
133    }
134    return returnValue;
135  }
136
137}
