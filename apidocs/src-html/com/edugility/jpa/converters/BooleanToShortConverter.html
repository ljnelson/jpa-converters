028package com.edugility.jpa.converters;
029
030import javax.persistence.AttributeConverter;
031import javax.persistence.Converter;
032
033/**
034 * An {@link AttributeConverter} that converts a {@link Boolean} to
035 * and from a {@link Short} representation.
036 *
037 * <p>This {@link AttributeConverter} is used mostly to map the Java
038 * {@link Boolean} type to a database type of {@code TINYINT},
039 * following the <a
040 * href="https://docs.oracle.com/javase/1.5.0/docs/guide/jdbc/getstart/mapping.html#1007274">recommened
041 * JDBC type mapping of {@code TINYINT} to {@code short}</a>.</p>
042 *
043 * @author <a href="http://about.me/lairdnelson"
044 * target="_parent">Laird Nelson</a>
045 *
046 * @see AttributeConverter
047 */
048@Converter
049public class BooleanToShortConverter implements AttributeConverter<Boolean, Short> {
050
051  /**
052   * Creates a new {@link BooleanToShortConverter}.
053   */
054  public BooleanToShortConverter() {
055    super();
056  }
057
058  /**
059   * Returns a {@link Short} representing the supplied {@link
060   * Boolean}.
061   *
062   * <p>This method may return {@code null}.</p>
063   *
064   * @param b the {@link Boolean} to convert; may be {@code null} in
065   * which case {@code null} will be returned
066   *
067   * @return {@code null} if {@code b} is {@code null}, {@link
068   * Short#valueOf(short) Short.valueOf((short)1)} if {@code b} is
069   * {@linkplain Boolean#equals(Object) equal to} {@link
070   * Boolean#TRUE}, and {@link Short#valueOf(short)
071   * Short.valueOf((short)0)} in all other cases
072   */
073  @Override
074  public Short convertToDatabaseColumn(final Boolean b) {
075    final Short returnValue;
076    if (b == null) {
077      returnValue = null;
078    } else if (Boolean.TRUE.equals(b)) {
079      returnValue = Short.valueOf((short)1);
080    } else {
081      returnValue = Short.valueOf((short)0);
082    }
083    return returnValue;
084  }
085
086  /**
087   * Returns a {@code Boolean} representing the supplied {@link
088   * Short}.
089   *
090   * <p>This method may return {@code null}.</p>
091   *
092   * <p>If the supplied {@link Short} is {@code null}, then {@code
093   * null} will be returned.</p>
094   *
095   * <p>If the supplied {@link Short}'s {@linkplain Short#shortValue()
096   * value} is greater than {@code 0}, then {@link Boolean#TRUE} will
097   * be returned.</p>
098   *
099   * <p>In all other cases, {@link Boolean#FALSE} will be
100   * returned.</p>
101   *
102   * @param s the {@link Short} to convert; may be {@code null} in
103   * which case {@code null} will be returned
104   *
105   * @return {@code null} if {@code s} is {@code null}, 
106   * {@link Boolean#TRUE} if {@code s}'s {@linkplain Short#shortValue()
107   * value} is greater than {@code 0}, or {@link Boolean#FALSE} in all
108   * other cases className}
109   */
110  @Override
111  public Boolean convertToEntityAttribute(final Short s) {
112    final Boolean returnValue;
113    if (s == null) {
114      returnValue = null;
115    } else if (s.shortValue() > 0) {
116      returnValue = Boolean.TRUE;
117    } else {
118      returnValue = Boolean.FALSE;
119    }
120    return returnValue;
121  }
122
123}
