028package com.edugility.jpa.converters;
029
030import java.util.Locale;
031
032import javax.persistence.AttributeConverter;
033
034import org.junit.Before;
035import org.junit.Test;
036
037import static org.junit.Assert.assertEquals;
038import static org.junit.Assert.assertFalse;
039import static org.junit.Assert.assertNotNull;
040import static org.junit.Assert.assertTrue;
041
042public class TestCaseLocaleRoundTripping {
043
044  private AttributeConverter<Locale, String> converter;
045
046  private Locale[] availableLocales;
047
048  public TestCaseLocaleRoundTripping() {
049    super();
050  }
051
052  @Before
053  public void setUp() {
054    this.converter = new LocaleToLanguageTagConverter();
055    this.availableLocales = Locale.getAvailableLocales();
056    assertNotNull(this.availableLocales);
057    assertTrue(this.availableLocales.length > 0);
058  }
059
060  @Test
061  public void testCaseLocaleRoundTripping() {
062    for (final Locale availableLocale : this.availableLocales) {
063      assertNotNull(availableLocale);
064      final String converterStringRepresentation = this.converter.convertToDatabaseColumn(availableLocale);
065      assertNotNull(converterStringRepresentation);
066      assertFalse(converterStringRepresentation.isEmpty());
067      final Locale converterLocaleRepresentation = this.converter.convertToEntityAttribute(converterStringRepresentation);
068      if (!availableLocale.equals(converterLocaleRepresentation) && !"no_NO_NY".equals(availableLocale.toString())) {
069        // See "Special Conversions" in http://docs.oracle.com/javase/8/docs/api/java/util/Locale.html#toLanguageTag--
070        assertEquals(availableLocale, converterLocaleRepresentation);
071      }
072    }
073  }
074
075}
